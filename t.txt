#include "disp.h"

union disp_data {
    struct {
        disp_val *car;
        disp_val *cdr;
    } cons;
};

GC_UNION_TI(disp_data,
    GC_OFF(disp_data, cons.car),
    GC_OFF(disp_data, cons.cdr)
);

// ========== cons data pool（bump allocator）==========

//#define CONS_POOL_SIZE  4096
#define CONS_POOL_SIZE  64

static union disp_data* g_cons_pool = NULL;
static union disp_data* g_cons_curr = NULL;
static gc_mutex_t *g_cons_lock;

// 初始化 cons 池（在 disp_init 中调用）
void disp_init_cons_pool(void) {
    if (g_cons_pool) return;
    gc_pthread_mutex_init(&g_cons_lock, NULL);
    gc_add_root(&g_cons_lock);
    gc_add_root(&g_cons_pool);
    g_cons_pool = gc_typed_calloc(sizeof(union disp_data), CONS_POOL_SIZE, &union_disp_data_ti);
    g_cons_curr = g_cons_pool;
}

static union disp_data* alloc_cons_data(void) {
    gc_pthread_mutex_lock(g_cons_lock);
    if (g_cons_curr >= g_cons_pool + CONS_POOL_SIZE) {
        gc_collect();
        g_cons_pool = gc_typed_calloc(sizeof(union disp_data), CONS_POOL_SIZE, &union_disp_data_ti);
        g_cons_curr = g_cons_pool;
    }

    union disp_data *result;
    result = g_cons_curr++;
    gc_pthread_mutex_unlock(g_cons_lock);
    return result;
}

disp_val* disp_make_cons(disp_val *car, disp_val *cdr) {
    union disp_data *data = alloc_cons_data();
    disp_val *v = disp_alloc(DISP_CONS, data);
    v->data->cons.car = car;
    v->data->cons.cdr = cdr;
    return v;
}

disp_val* disp_car(disp_val *cons) {
    return (cons && cons->flag == DISP_CONS) ? cons->data->cons.car : NIL;
}

disp_val* disp_cdr(disp_val *cons) {
    return (cons && cons->flag == DISP_CONS) ? cons->data->cons.cdr : NIL;
}

void disp_set_car(disp_val *cons, disp_val *car) {
    if (cons && cons->flag == DISP_CONS) GC_ASSIGN_PTR(cons->data->cons.car, car);
}

void disp_set_cdr(disp_val *cons, disp_val *cdr) {
    if (cons && cons->flag == DISP_CONS) GC_ASSIGN_PTR(cons->data->cons.cdr, cdr);
}
